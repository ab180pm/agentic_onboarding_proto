Handle Deep Link in SceneDelegate
import UIKit
import Airbridge

class SceneDelegate: UIResponder, UIWindowSceneDelegate {

    // Deep link handler
    func handleAirbridgeDeeplink(url: URL) {
        // Navigate to specific screen based on URL
    }

    // Cold start with Universal Link
    func scene(_ scene: UIScene, willConnectTo session: UISceneSession,
               options connectionOptions: UIScene.ConnectionOptions) {
        // Handle userActivities
        if let userActivity = connectionOptions.userActivities.first {
            Airbridge.trackDeeplink(userActivity) { url in
                self.handleAirbridgeDeeplink(url: url)
            }
        }
        // Handle urlContexts
        if let urlContext = connectionOptions.urlContexts.first {
            Airbridge.trackDeeplink(urlContext.url) { url in
                self.handleAirbridgeDeeplink(url: url)
            }
        }
    }

    // Scheme deep link (app in background)
    func scene(_ scene: UIScene,
               openURLContexts URLContexts: Set<UIOpenURLContext>) {
        guard let urlContext = URLContexts.first else { return }
        Airbridge.trackDeeplink(urlContext.url) { url in
            self.handleAirbridgeDeeplink(url: url)
        }
    }

    // Universal Link (app in background)
    func scene(_ scene: UIScene,
               continue userActivity: NSUserActivity) {
        Airbridge.trackDeeplink(userActivity) { url in
            self.handleAirbridgeDeeplink(url: url)
        }
    }
}
